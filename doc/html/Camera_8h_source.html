90template<typename M, typename E> std::shared_ptr<Vector3D<M, E>> Camera<M, E>::getE1() const {
91 std::shared_ptr<Vector3D<M, E>> pv;
92 if (ppoint!=NULL && pnormal!=NULL){
93 pv=std::make_shared<Vector3D<M, E>>(
94 ppoint, Point3D<M, E>{
95 {1,0},
96 {1,0},
97 {(M)(-1.)/pnormal->pp2->z * (pnormal->pp2->x*((M)(1.)-ppoint->x) + pnormal->pp2->y*((M)(1.)-ppoint->y)) + ppoint->z}
98 }
99 );//One possible normal vector
100 *pv/=pv->getNorm();//Normalisation
101 }
102 return pv;
103}

136template<typename M, typename E> bool Camera<M, E>::testInFielOfView(const Point3D<M, E>& p) const {
137 bool test=false;
138 if (ppoint!=NULL){
139 Vector3D<M, E> vpos(ppoint, p-*ppoint);
140
141 if (m_aperture/2. >= acos( ((*pnormal*vpos) / (pnormal->getNorm()*vpos.getNorm())).to_m_type() )){
142 test=true;
143 }
144 }
