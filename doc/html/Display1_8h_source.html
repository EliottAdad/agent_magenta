222template<typename T> bool Display1<T>::render(const std::shared_ptr<Point3D<T>> ppoint) const {
223 bool success=false;
224 //printf("Rendering point\n");
225 if (this->ppoint!=NULL){
226 Point3D<T> d_point=*ppoint-*(this->ppoint);
227
228 if (pwindow!=NULL && prenderer!=NULL && pdraw_color!=NULL){
229 changeRenderDrawColor(prenderer, pdraw_color);
230 //SDL_SetRenderDrawColor(prenderer.get(), pdraw_color->r, pdraw_color->g, pdraw_color->b, pdraw_color->a); // Chooses the draw color.
231
232 // Get the dimensions of the window
233 int sizex(0);
234 int sizey(0);
235 SDL_GetWindowSize(pwindow.get(), &sizex, &sizey);
236 T centerx=(T)(sizex/2);
237 T centery=(T)(sizey/2);
238
239 if (prenderer!=NULL){
240 switch (m_display){
241 case 1://(x,y) plane
242 //printf("\nx=%i\n", (int)(d_point.x*(M)m_scale + centerx).to_m_type());
243 drawPointRenderer(prenderer, (int)(d_point.x*m_scale + centerx).to_m_type(), (int)(d_point.y*m_scale + centery).to_m_type());
244 break;
245 case 2://(y,z) plane
246 drawPointRenderer(prenderer, (int)(d_point.y*m_scale + centerx).to_m_type(), (int)(d_point.z*m_scale + centery).to_m_type());
247 break;
248 case 3://(x,z) plane
249 drawPointRenderer(prenderer, (int)(d_point.x*m_scale + centerx).to_m_type(), (int)(d_point.z*m_scale + centery).to_m_type());
250 break;
251 }
252 }
253 success=true;
254 }
255 }
256 return success;
